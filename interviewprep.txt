const users = [
  { name: "Alice", age: 25 },
  { name: "Bob", age: 16 },
  { name: "Charlie", age: 30 },
];

function getage(usernames, userage) {
  return usernames.filter((elem) => elem.age < userage);
}
console.log(getage(users, 18));

6.use foreach loop find the greatest one 


const users = [
  { name: "Alice", age: 25 },
  { name: "Bob", age: 16 },
  { name: "Charlie", age: 30 },
];

let oldest = users[0];
users.forEach((elem) => {
  if (elem.age > oldest.age) {
    oldest = elem;
  }
});

console.log(oldest)
